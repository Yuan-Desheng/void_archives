当多个组件都需要「发请求 + 记录 loading/error」时，把这段逻辑抽成一个以 use 开头的函数，就是自定义 Hook。

下面的 useFetch 内部复用了 useState 和 useEffect，对外只暴露 { data, loading, error }，调用方完全不用关心请求细节。

function useFetch(url) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  useEffect(() => {
    fetch(url).then(r => r.json()).then(setData).finally(() => setLoading(false));
  }, [url]);
  return { data, loading };
}
